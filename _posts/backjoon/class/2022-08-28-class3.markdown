{% highlight python %}
import sys
input = sys.stdin.readline

N, M = map(int,input().split())

n_named = set()
for _ in range(N):
    name = input().rstrip()
    n_named.add(name)

m_named = set()
for _ in range(M):
    name = input().rstrip()
    m_named.add(name)

lists = list(n_named&m_named)
result = sorted(lists)

print(len(result))
for r in result:
    print(r)
