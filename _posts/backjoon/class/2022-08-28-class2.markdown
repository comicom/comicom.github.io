## 팰린드롬수

{% highlight python %}
import sys
input = sys.stdin.readline

lists = []
string = input().rstrip("\n")
lists.append(string)
while string != '0':
    string = input().rstrip("\n")
    lists.append(string)

lists = lists[:-1]
result = ['']*len(lists)
for i in range(len(lists)):
    result[i] = "yes"
    for j in range(0,len(lists[i])//2):
        if lists[i][j] != lists[i][-j-1]:
            result[i] = "no"
            break

for r in result:
    print(r)
{% endhighlight %}
