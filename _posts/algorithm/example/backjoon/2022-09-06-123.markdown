## 1, 2, 3 더하기

{% highlight python %}
import sys
input = sys.stdin.readline

# 1, 1
# 2, 11, 2
# 3, 111, 12, 21, 3
# 4, 1111, 112, 121, 211, 22, 13, 31

dp = [0]*12
dp[1] = 1
dp[2] = 2
dp[3] = 4

def find123(dp,num):
    if dp[num] != 0:
        return dp[num]
    else:
        dp[num] = find123(dp,num-1) + find123(dp,num-2) + find123(dp,num-3)
        return dp[num]

T = int(input())
for _ in range(T):
    num = int(input())
    print(find123(dp,num))
{% endhighlight %}
